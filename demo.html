class SaudiIDValidator:
    """
    Hardened Saudi National ID/Iqama Validation
    - Multi-stage regex extraction
    - Modified Luhn checksum validation
    - Zero false-positives on phone/invoice numbers
    """
    
    def validate_checksum(self, normalized_id: str) -> bool:
        if len(normalized_id) != 10:
            return False
        
        digits = [int(d) for d in normalized_id]
        checksum = 0
        
        for idx, digit in enumerate(reversed(digits)):
            if idx % 2 == 1:
                doubled = digit * 2
                checksum += doubled - 9 if doubled > 9 else doubled
            else:
                checksum += digit
                
        return checksum % 10 == 0

⚙️ كيف يعمل النظام؟

1️⃣ المعالجة المحلية الكاملة

جميع العمليات (استخراج الجذور، كشف الهويات، التحليل) تحدث في متصفحك فقط. لا يتم إرسال أي بيانات لخوادم خارجية.

2️⃣ التحقق الرياضي (Checksum Validation)

نستخدم خوارزمية Luhn المعدلة للتحقق من صحة أرقام الهوية والإقامات، مما يمنع الإيجابيات الخاطئة على أرقام الجوال والفواتير.

3️⃣ حماية متعددة الطبقات (XOA Isolation)

نظام عزل بثلاثة مستويات (Ring-0/1/2) يمنع البيانات غير الموثوقة من الوصول لنظام التحكم الأساسي.

4️⃣ توافق تنظيمي (CST Class C)

مصمم للامتثال الكامل لمتطلبات هيئة الاتصالات السعودية للبيانات الحساسة والحكومية.
